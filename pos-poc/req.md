### иіҮж–ҷиЎЁзөҗж§Ӣ

#### з§ҹжҲ¶/е“ҒзүҢ (tenants)

```sql
CREATE TABLE tenants (
    id UUID PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    tax_id VARCHAR(20),              -- зөұдёҖз·Ёиҷҹ
    invoice_config JSONB,            -- йӣ»еӯҗзҷјзҘЁиЁӯе®ҡ
    created_at TIMESTAMPTZ DEFAULT NOW()
);
```

#### еҲҶеә— (stores)

```sql
CREATE TABLE stores (
    id UUID PRIMARY KEY,
    tenant_id UUID REFERENCES tenants(id),
    name VARCHAR(100) NOT NULL,
    address TEXT,
    phone VARCHAR(20),
    is_active BOOLEAN DEFAULT true,
    settings JSONB,                  -- еҲҶеә—иЁӯе®ҡ (зҮҹжҘӯжҷӮй–“зӯү)
    created_at TIMESTAMPTZ DEFAULT NOW()
);
```

#### е“Ўе·Ҙ (employees)

```sql
CREATE TABLE employees (
    id UUID PRIMARY KEY,
    tenant_id UUID REFERENCES tenants(id),
    store_id UUID REFERENCES stores(id),
    name VARCHAR(50) NOT NULL,
    pin_code VARCHAR(10),            -- еҝ«йҖҹзҷ»е…Ҙзўј
    role VARCHAR(20),                -- admin, manager, cashier, kitchen
    permissions JSONB,
    is_active BOOLEAN DEFAULT true
);
```

#### иҸңе–®еҲҶйЎһ (menu_categories)

```sql
CREATE TABLE menu_categories (
    id UUID PRIMARY KEY,
    tenant_id UUID REFERENCES tenants(id),
    name VARCHAR(50) NOT NULL,
    sort_order INT DEFAULT 0,
    is_active BOOLEAN DEFAULT true
);
```

#### е•Ҷе“Ғ (menu_items)

```sql
CREATE TABLE menu_items (
    id UUID PRIMARY KEY,
    tenant_id UUID REFERENCES tenants(id),
    category_id UUID REFERENCES menu_categories(id),
    name VARCHAR(100) NOT NULL,
    description TEXT,
    price DECIMAL(10,2) NOT NULL,
    cost DECIMAL(10,2),              -- жҲҗжң¬
    image_url TEXT,
    barcode VARCHAR(50),             -- жўқзўј
    is_active BOOLEAN DEFAULT true,
    options JSONB,                   -- еҠ ж–ҷгҖҒе®ўиЈҪеҢ–йҒёй …
    tax_type VARCHAR(20),            -- зЁ…еҲҘ
    sort_order INT DEFAULT 0
);
```

#### е•Ҷе“ҒеҲҶеә—еғ№ж ј (menu_item_prices)

```sql
CREATE TABLE menu_item_prices (
    id UUID PRIMARY KEY,
    item_id UUID REFERENCES menu_items(id),
    store_id UUID REFERENCES stores(id),
    price DECIMAL(10,2) NOT NULL,
    UNIQUE(item_id, store_id)
);
```

#### жЎҢдҪҚ (tables)

```sql
CREATE TABLE tables (
    id UUID PRIMARY KEY,
    store_id UUID REFERENCES stores(id),
    name VARCHAR(20) NOT NULL,       -- жЎҢиҷҹ
    capacity INT,
    area VARCHAR(50),                -- еҚҖеҹҹ (е®Өе…§гҖҒжҲ¶еӨ–)
    status VARCHAR(20) DEFAULT 'available'  -- available, occupied, reserved
);
```

#### иЁӮе–® (orders)

```sql
CREATE TABLE orders (
    id UUID PRIMARY KEY,
    store_id UUID REFERENCES stores(id),
    order_no VARCHAR(30) NOT NULL,          -- е–®иҷҹ (жҜҸж—ҘжөҒж°ҙиҷҹ)
    order_type VARCHAR(20) NOT NULL,        -- dine_in, takeout, delivery
    table_id UUID REFERENCES tables(id),
    customer_name VARCHAR(50),
    customer_phone VARCHAR(20),
    delivery_address TEXT,
    delivery_platform VARCHAR(20),          -- uber_eats, foodpanda, иҮӘе®¶еӨ–йҖҒ
    
    subtotal DECIMAL(10,2),
    discount DECIMAL(10,2) DEFAULT 0,
    tax DECIMAL(10,2),
    service_charge DECIMAL(10,2) DEFAULT 0,
    total DECIMAL(10,2),
    
    status VARCHAR(20) DEFAULT 'pending',   -- pending, preparing, ready, completed, cancelled
    payment_status VARCHAR(20) DEFAULT 'unpaid',
    
    employee_id UUID REFERENCES employees(id),
    notes TEXT,
    
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);
```

#### иЁӮе–®жҳҺзҙ° (order_items)

```sql
CREATE TABLE order_items (
    id UUID PRIMARY KEY,
    order_id UUID REFERENCES orders(id),
    item_id UUID REFERENCES menu_items(id),
    item_name VARCHAR(100),          -- еҝ«з…§
    unit_price DECIMAL(10,2),
    quantity INT NOT NULL,
    subtotal DECIMAL(10,2),
    options JSONB,                   -- йҒёж“Үзҡ„еҠ ж–ҷ
    notes TEXT,                      -- еӮҷиЁ» (еҺ»еҶ°гҖҒе°‘зі–)
    status VARCHAR(20) DEFAULT 'pending'   -- pending, preparing, ready
);
```

#### д»ҳж¬ҫзҙҖйҢ„ (payments)

```sql
CREATE TABLE payments (
    id UUID PRIMARY KEY,
    order_id UUID REFERENCES orders(id),
    method VARCHAR(20) NOT NULL,     -- cash, credit_card, line_pay
    amount DECIMAL(10,2) NOT NULL,
    received DECIMAL(10,2),          -- ж”¶ж¬ҫйҮ‘йЎҚ (зҸҫйҮ‘з”Ё)
    change DECIMAL(10,2),            -- жүҫйӣ¶
    reference_no VARCHAR(100),       -- дәӨжҳ“еәҸиҷҹ
    status VARCHAR(20) DEFAULT 'completed',
    created_at TIMESTAMPTZ DEFAULT NOW()
);
```

#### йӣ»еӯҗзҷјзҘЁ (invoices)

```sql
CREATE TABLE invoices (
    id UUID PRIMARY KEY,
    order_id UUID REFERENCES orders(id),
    invoice_no VARCHAR(20),          -- зҷјзҘЁиҷҹзўј
    random_code VARCHAR(4),          -- йҡЁж©ҹзўј
    buyer_tax_id VARCHAR(10),        -- иІ·ж–№зөұз·Ё
    amount DECIMAL(10,2),
    tax DECIMAL(10,2),
    status VARCHAR(20),              -- pending, issued, voided
    carrier_type VARCHAR(20),        -- ијүе…·йЎһеһӢ
    carrier_no VARCHAR(50),          -- ијүе…·иҷҹзўј
    issued_at TIMESTAMPTZ,
    raw_response JSONB
);
```

#### еә«еӯҳ (inventory)

```sql
CREATE TABLE inventory (
    id UUID PRIMARY KEY,
    store_id UUID REFERENCES stores(id),
    item_id UUID REFERENCES menu_items(id),
    quantity DECIMAL(10,2),
    unit VARCHAR(20),
    low_stock_threshold DECIMAL(10,2),
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(store_id, item_id)
);
```

// pkg/hardware/printer/receipt.go

func (p *Printer) PrintReceipt(order *domain.Order) error {
    var buf bytes.Buffer
    
    // еҲқе§ӢеҢ–
    buf.WriteString(Init)
    
    // еә—еҗҚ (зҪ®дёӯгҖҒж”ҫеӨ§)
    buf.WriteString(AlignCenter)
    buf.WriteString(FontDouble)
    buf.WriteString(order.StoreName + "\n")
    buf.WriteString(FontNormal)
    buf.WriteString(order.StoreAddress + "\n")
    buf.WriteString(order.StorePhone + "\n")
    buf.WriteString("--------------------------------\n")
    
    // иЁӮе–®иіҮиЁҠ
    buf.WriteString(AlignLeft)
    buf.WriteString(fmt.Sprintf("е–®иҷҹ: %s\n", order.OrderNo))
    buf.WriteString(fmt.Sprintf("жҷӮй–“: %s\n", order.CreatedAt.Format("2006-01-02 15:04")))
    buf.WriteString(fmt.Sprintf("йЎһеһӢ: %s\n", order.OrderType))
    if order.TableName != "" {
        buf.WriteString(fmt.Sprintf("жЎҢиҷҹ: %s\n", order.TableName))
    }
    buf.WriteString("--------------------------------\n")
    
    // е“Ғй …
    for _, item := range order.Items {
        buf.WriteString(fmt.Sprintf("%-16s x%d  $%d\n", 
            item.Name, item.Quantity, item.Subtotal))
        if item.Notes != "" {
            buf.WriteString(fmt.Sprintf("  (%s)\n", item.Notes))
        }
    }
    buf.WriteString("--------------------------------\n")
    
    // йҮ‘йЎҚ
    buf.WriteString(AlignRight)
    buf.WriteString(fmt.Sprintf("е°ҸиЁҲ: $%d\n", order.Subtotal))
    if order.Discount > 0 {
        buf.WriteString(fmt.Sprintf("жҠҳжүЈ: -$%d\n", order.Discount))
    }
    buf.WriteString(FontDouble)
    buf.WriteString(fmt.Sprintf("зёҪиЁҲ: $%d\n", order.Total))
    buf.WriteString(FontNormal)
    
    // д»ҳж¬ҫиіҮиЁҠ
    buf.WriteString(fmt.Sprintf("д»ҳж¬ҫ: %s\n", order.PaymentMethod))
    if order.PaymentMethod == "cash" {
        buf.WriteString(fmt.Sprintf("ж”¶ж¬ҫ: $%d\n", order.Received))
        buf.WriteString(fmt.Sprintf("жүҫйӣ¶: $%d\n", order.Change))
    }
    buf.WriteString("--------------------------------\n")
    
    // зҷјзҘЁиіҮиЁҠ
    buf.WriteString(AlignCenter)
    buf.WriteString(fmt.Sprintf("зҷјзҘЁиҷҹзўј: %s\n", order.InvoiceNo))
    buf.WriteString(fmt.Sprintf("йҡЁж©ҹзўј: %s\n", order.RandomCode))
    
    // и¬қи¬қе…үиҮЁ
    buf.WriteString("\nи¬қи¬қе…үиҮЁ\n\n")
    
    // еҲҮзҙҷ
    buf.WriteString(CutPaper)
    
    return p.Write(buf.Bytes())
}
